#include <stdio.h>

int main() {
    int n, i;
    
    float sum = 0.0, avg;
    
    printf("Enter the number of elements: ");
    
    if (scanf("%d", &n) != 1 || n <= 0) {
        printf("Please enter a valid positive number.\n");
        return 1;
    }

    int arr[n]; 


    printf("Enter %d elements:\n", n);
    for(i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
        sum += arr[i]; 
    }

    
    avg = sum / n; 

    
    printf("\nSum = %.2f\n", sum);
    printf("Average = %.2f\n", avg);

    return 0;
}
